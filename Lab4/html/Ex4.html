Equation: x + ln(x) = 0

f = @(x) x + log(x);

%Sabem que hi te una arrel al voltant de 0.5
t = 0:0.01:1;
%zeros per dibuixar la linea de 0 i 'm' pel color
plot(t, f(t), t, zeros(size(t)),'m'), grid, title('Equation');

Fzero Matlab

alpha = fzero(f,1)

   0.567143290409784

Iterative method #1

g1 = @(x)-log(x);
dg1 = @(x)-1./x;
x0 = 0.5;
tol = 0.0005;
N = 10;
if (abs(dg1(x0)) < 1)
    [ root, x_sol ] = new_fixPoint(f,g1,x0,tol,N)
    tolf = f(x_sol)
else
    disp('Divergent!')
end
