Iterative method #1

g1 = @(x)x.^6-1;
dg1 = @(x)6*x.^5;
x0 = c;
tol = 0.0005;
N = 10;
%We need to see if it's divergent or convergent. If df(x0) < 1 convergent,
%divergent otherwise
if (abs(dg1(x0)) < 1)
    [ root, x_sol ] = new_fixPoint(f,g1,x0,tol,N)
    tolf = f(x_sol)
else
    disp('Divergent!')
end

Divergent!

Iterative method #2

g2 = @(x)(x+1).^(1/6); % diferent f
dg2 = @(x) 1./(x+1).^(5/6)/6;
x0 = c;
tol = 0.0005;
N = 10;

if (abs(dg2(x0)) < 1)
    [ root, x_sol ] = new_fixPoint(f,g2,x0,tol,N)
    tolf = f(x_sol)
else
    disp('Divergent!')
end

    0.7860    0.0596    0.0052    0.0005

Estudi convergència

y=ones(size(t)); %Simplement dos linies a 1, per veure allo del if de si esta dins
